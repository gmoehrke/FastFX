1 /*
2  * Definition of FXController.
3  *
4  * FXController - is the primary class used for displaying/running effects/colors/etc. Initialize with a PixelController
5  * like: new FXController( new FastLEDPixelController( ledBufferPtr, numberOfLEDs ) );
6  *
7  * Effects/Colors are placed into FXSegments which are managed by the FXController. Every controller has a "Primary" segment, which
8  * represents the entire strip/array. This segment is always present and does not need to be created manually.
9  *
10  * Additional (overlay) segments are added "on top" of the primary segment by calling AddSegment(...). Each of these overlay
11  * segments, can have its own effect running and has independent opacity/transparency and brightness. Each additional segment
12  * is referenced by a unique "Tag" or Name, which is used to address and manipulate the segment.
13  *
14  * Displaying and changing effects (Primary Segment):
15  *
16  * FX = new RainbowFX( numberOfLEDs, updateInterval );
17  * FXController->getPrimarySegment()->setFX( FX );
18  *
19  * Displaying and changing effects (Additional Segments)
20  *
21  * FXController->AddSegment( "Left", 0, 25, null )
22  * FXSegment *seg = FXController->findSegment("Left");
23  * if (seg) {
24  * seg->setFX( new RainbowFX(seg->getLength()));
25  * set->setOpacity(255);
26  * }
27  *
28  * ** This will show the rotating rainbow effect on pixels 0-25 and the remainder of the strip will continue to show the primary effect.
29  *
30  * inside main processing loop:
31  *
32  * FFXController.update();
33  *
34  * Each FFXSegment automatically supports cross-fading by using an independent FXFrameProvider. The Frame Provider does this by allocating extra buffers and using cycles
35  * between refreshes to fade from one frame to the next. Turn this on and off using FXController->getPrimarySegment()->getFrameProvider()->setFade( boolean ) or
36  * FXController->findSegment("name")->getFrameProvider()->setFade( boolean )
37  *
38  * Overlay FX are effects that can be displayed over the top of existing/running effects (the entire array), such as flashes, pulses,
39  * etc. These are also cross-faded and blended with the underlying FX as desired. Both the overlay and the underlying
40  * effect are cross-faded and continue to animate while displaying. These are done as follows:
41  *
42  * FFXController.setFX( FX );
43  * OFX = new PulseOveralyFX( numberOfLEDs, speed );
44  * OFX.getFXColor().setPalette( myPal );
45  * FFXController.setOverlayFX( OFX );
46  *
47  * Note that FFXController() automatically deletes the overlay FX, when it is complete so it MUST be created using - new OverlayFX(...);
48  * See FXOverlay.h for more info.
49  */
