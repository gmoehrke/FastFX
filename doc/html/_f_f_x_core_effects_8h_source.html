525 
526  void mapPixels(CRGB *bufLeds)
527  {
528  for( uint16_t i = 0; i < numLeds; i++) {
529  if( pixelState[i] == SteadyDim) {
530  // this pixels is currently: SteadyDim
531  // so we randomly consider making it start getting brighter
532  if( random8() < chanceOfTwinkle) {
533  pixelState[i] = GettingBrighter;
534  }
535 
536  } else if( pixelState[i] == GettingBrighter ) {
537  // this pixels is currently: GettingBrighter
538  // so if it's at peak color, switch it to getting dimmer again
539  if( bufLeds[i] >= peakColor ) {
540  pixelState[i] = GettingDimmerAgain;
541  } else {
542  // otherwise, just keep brightening it:
543  bufLeds[i] += CRGB( random8(deltaColorUp.r), random8(deltaColorUp.g), random8(deltaColorUp.b) );
544  }
545  } else { // getting dimmer again
546  // this pixels is currently: GettingDimmerAgain
547  // so if it's back to base color, switch it to steady dim
548  if( bufLeds[i] == baseColor ) {
549  // bufLeds[i] = baseColor; // reset to exact base color, in case we overshot

600 class PacificaFX : public FFXBase {
601 //
602 // This lighting effect called "Pacifica" was specially created to feel like moving light on water and to look like light moving under water.
603 // Programmer Mark Kriegsman created it on the FASTLED platform in consultation and collaboration with Mary March for her art work
604 // titled "Beneath These Waves Lies Light".
605 //
607 //
608 // In this animation, there are four "layers" of waves of light.
609 //
610 // Each layer moves independently, and each is scaled separately.
611 //
612 // All four wave layers are added together on top of each other, and then
613 // another filter is applied that adds "whitecaps" of brightness where the
614 // waves line up with each other more. Finally, another pass is taken
615 // over the led array to 'deepen' (dim) the blues and greens.
616 //
617 // The speed and scale and motion each layer varies slowly within independent
618 // hand-chosen ranges, which is why the code has a lot of low-speed 'beatsin8' functions
619 // with a lot of oddly specific numeric ranges.
620 //
621 // These three custom blue-green color palettes were inspired by the colors found in
622 // the waters off the southern coast of California, https://goo.gl/maps/QQgd97jjHesHZVxQ7
623 //
624 // Modified from original FastLED native sketch foound here: https://gist.github.com/kriegsman/36a1e277f5b4084258d9af1eae29bac4
625 //
626 public:
627  PacificaFX( uint16_t initSize ) : FFXBase( initSize, (uint8_t)255, 10, 20 ) {
628  fxid = PACIFICA_FX_ID;
629  fxName = PACIFICA_FX_NAME;
630  }
631 
632  virtual void writeNextFrame( CRGB *bufLeds ) override {
633  pacifica_loop( bufLeds );
634  }
635 
636 private:
637  void pacifica_loop( CRGB *bufLeds ) {
638  // Increment the four "color index start" counters, one for each wave layer.
639  // Each is incremented at a different speed, and the speeds vary over time.
640  uint32_t ms = millis();
641  uint32_t deltams = ms - sLastms;
642  sLastms = ms;
643  uint16_t speedfactor1 = beatsin16(3, 179, 269);
644  uint16_t speedfactor2 = beatsin16(4, 179, 269);
645  uint32_t deltams1 = (deltams * speedfactor1) / 256;
646  uint32_t deltams2 = (deltams * speedfactor2) / 256;
647  uint32_t deltams21 = (deltams1 + deltams2) / 2;
648  sCIStart1 += (deltams1 * beatsin88(1011,10,13));
649  sCIStart2 -= (deltams21 * beatsin88(777,8,11));
650  sCIStart3 -= (deltams1 * beatsin88(501,5,7));
651  sCIStart4 -= (deltams2 * beatsin88(257,4,6));
652 
653  // Clear out the LED array to a dim background blue-green
654  fill_solid( bufLeds, numLeds, CRGB( 2, 6, 10));
655 
656  // Render each of four layers, with different scales and speeds, that vary over time
657  pacifica_one_layer( bufLeds, pacifica_palette_1, sCIStart1, beatsin16( 3, 11 * 256, 14 * 256), beatsin8( 10, 70, 130), 0-beat16( 301) );
658  pacifica_one_layer( bufLeds, pacifica_palette_2, sCIStart2, beatsin16( 4, 6 * 256, 9 * 256), beatsin8( 17, 40, 80), beat16( 401) );
659  pacifica_one_layer( bufLeds, pacifica_palette_3, sCIStart3, 6 * 256, beatsin8( 9, 10,38), 0-beat16(503));
660  pacifica_one_layer( bufLeds, pacifica_palette_3, sCIStart4, 5 * 256, beatsin8( 8, 10,28), beat16(601));
661 
662  // Add brighter 'whitecaps' where the waves lines up more
663  pacifica_add_whitecaps(bufLeds);
664 
665  // Deepen the blues and greens a bit
666  pacifica_deepen_colors(bufLeds);
667  }
668 
669  // Add one layer of waves into the led array
670  void pacifica_one_layer( CRGB *bufLeds,const CRGBPalette16& p, uint16_t cistart, uint16_t wavescale, uint8_t bri, uint16_t ioff) {
671  uint16_t ci = cistart;
672  uint16_t waveangle = ioff;
673  uint16_t wavescale_half = (wavescale / 2) + 20;
674  for( uint16_t i = 0; i < numLeds; i++) {
675  waveangle += 250;
676  uint16_t s16 = sin16( waveangle ) + 32768;
677  uint16_t cs = scale16( s16 , wavescale_half ) + wavescale_half;
678  ci += cs;
679  uint16_t sindex16 = sin16( ci) + 32768;
680  uint8_t sindex8 = scale16( sindex16, 240);
681  CRGB c = ColorFromPalette( p, sindex8, bri, LINEARBLEND);
682  bufLeds[i] += c;
683  }
684  }
685 
686  // Add extra 'white' to areas where the four layers of light have lined up brightly
687  void pacifica_add_whitecaps( CRGB *bufLeds ) {
688  uint8_t basethreshold = beatsin8( 9, 55, 65);
689  uint8_t wave = beat8( 7 );
690 
691  for( uint16_t i = 0; i < numLeds; i++) {
692  uint8_t threshold = scale8( sin8( wave), 20) + basethreshold;
693  wave += 7;
694  uint8_t l = bufLeds[i].getAverageLight();
695  if( l > threshold) {
696  uint8_t overage = l - threshold;
697  uint8_t overage2 = qadd8( overage, overage);
698  bufLeds[i] += CRGB( overage, overage2, qadd8( overage2, overage2));
699  }
700  }
701  }
702 
703  // Deepen the blues and greens
704  void pacifica_deepen_colors( CRGB *bufLeds)
705  {
706  for( uint16_t i = 0; i < numLeds; i++) {
707  bufLeds[i].blue = scale8( bufLeds[i].blue, 145);
708  bufLeds[i].green= scale8( bufLeds[i].green, 200);
709  bufLeds[i] |= CRGB( 2, 5, 7);
710  }
711  }
712 
713  private:
714  uint16_t sCIStart1, sCIStart2, sCIStart3, sCIStart4;
715  uint32_t sLastms = 0;
716 
717 };
